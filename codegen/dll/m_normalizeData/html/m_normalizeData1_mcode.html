1function m_normalizeData(hObject)
 2    %this will run the normalization
 3    
 4    % FIXME! Print to logger instead
 5    % message = sprintf('Running normalization ...');
 6    %statusBarMessage(hObject, message, 0, false);
 7    
 8    % FIXME! Pass concrete type
 9    %handles = guidata(hObject);    
10    %ask where to create the normalized data folder
11    %path = handles.path;
12    % FIXME! Pass file names into the function! Don't retrieve them here!
13    % outside it.
14    %folderName = uigetdir(path, 'Where do you want to create the Normalized data folder?');
15    %if folderName == 0
16    %    return %we did not select a folder
17    %end
18    
19    %folderName = [folderName '/Normalized'];
20    %[~, ~] = mkdir(folderName);
21    
22    %listDataFiles = handles.files.fileNames;
23    %nbrFiles = numel(listDataFiles);
24    
25    [roiLogicalArray, isWithRoi] = getRoiLogicalArray(hObject);
26    
27    %take average of all open beam
28    averageTotalOpenBeam = getAverageArray(hObject, 'OB');
29    if isWithRoi
30        averageRoiOpenBeam = getAverageOfRoi(averageTotalOpenBeam, roiLogicalArray);
31    end
32    
33    %take average of all dark field
34    averageTotalDarkField = getAverageArray(hObject, 'DF');
35    data = handles.files.images;
36    
37    for i=1:nbrFiles
38     
39        %Don't use cell arrays
40    %    tmpData = data{i};
41        tmpDataFiltered = applyGammaFiltering(hObject, tmpData);
42        
43        imshow(tmpDataFiltered,[]);
44        colorbar;
45        
46        if isWithRoi
47            averageTmpData = getAverageOfRoi(tmpDataFiltered, roiLogicalArray);
48            ratio = averageTmpData / averageRoiOpenBeam;
49            tmpDataFiltered = tmpDataFiltered / ratio;
50        end
51        
52        topRatio = tmpDataFiltered - averageTotalDarkField;
53        bottomRatio = averageTotalOpenBeam - averageTotalDarkField;
54        
55        normalizedImage = topRatio ./ bottomRatio;
56        
57        %bring to zero all the counts < 0
58        lessThanZero = normalizedImage < 0;
59        normalizedImage(lessThanZero) = 0;
60        
61        %bring to 1 all the counts > 1
62        moreThanOne = normalizedImage > 1;
63        normalizedImage(moreThanOne) = 1;
64        
65        %normalizedImage = cast(normalizedImage,'int16');
66        
67        createNormalizedFile(normalizedImage, folderName, listDataFiles{i});
68        
69    end
70    
71    message = sprintf('Normalization is Done !');
72    statusBarMessage(hObject, message, 5, false);
73    
74end
